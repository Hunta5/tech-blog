DAY1-3
AI Agent 核心原理与手写实现
先不用框架——从零理解 Agent 是怎么"自主思考"的
🧠 Agent vs 普通 API 调用 必做
普通 API 调用AI Agent
流程你写死每一步AI 自己决定执行什么
工具你手动调用AI 自动选择调用哪个工具
循环调一次就完循环执行直到任务完成
容错你处理异常AI 观察错误后自己调整策略
类比遥控车自动驾驶
🔧 从零手写一个 ReAct Agent 必做

ReAct（Reasoning + Acting）是最经典的 Agent 模式。AI 先推理（Thought），再行动（Action），然后观察结果（Observation），循环执行直到完成。

react_agent.py — 手写 ReAct Agent
import json, re
from openai import OpenAI
from datetime import datetime

client = OpenAI()

# ========== 1. 工具定义 ==========
def search_web(query):
    """模拟网络搜索"""
    fake_results = {
        "Python": "Python 是一种高级编程语言，最新版本 3.12。",
        "RAG": "RAG 是检索增强生成，2024-2025 年最热门的 AI 技术之一。",
    }
    for key, val in fake_results.items():
        if key.lower() in query.lower():
            return val
    return f"未找到关于'{query}'的搜索结果"

def calculator(expression):
    """安全计算数学表达式"""
    try:
        return str(eval(expression, {"__builtins__": {}}, {}))
    except:
        return "计算错误"

def get_current_time():
    """获取当前时间"""
    return datetime.now().strftime("%Y-%m-%d %H:%M:%S")

TOOLS = {
    "search_web": {"func": search_web, "desc": "搜索网络信息，参数: query"},
    "calculator": {"func": calculator, "desc": "计算数学表达式，参数: expression"},
    "get_current_time": {"func": get_current_time, "desc": "获取当前时间，无参数"},
}

# ========== 2. Agent 系统 Prompt ==========
AGENT_PROMPT = """你是一个智能助手，可以使用工具完成任务。

可用工具：
{tool_descriptions}

工作流程：
1. 分析用户需求
2. 如果需要，调用工具获取信息
3. 根据结果回答问题

调用工具时，严格使用以下 JSON 格式（独占一行）：
TOOL_CALL: {{"tool": "工具名", "args": {{"参数名": "参数值"}}}}

如果不需要工具，直接回答。
如果任务已完成，以 FINAL_ANSWER: 开头给出最终回答。"""

# ========== 3. Agent 循环 ==========
class SimpleAgent:
    def __init__(self, tools, max_steps=5):
        self.tools = tools
        self.max_steps = max_steps

        tool_desc = "\n".join(
            [f"- {name}: {t['desc']}" for name, t in tools.items()]
        )
        self.system_prompt = AGENT_PROMPT.format(tool_descriptions=tool_desc)

    def run(self, task):
        messages = [
            {"role": "system", "content": self.system_prompt},
            {"role": "user", "content": task}
        ]

        for step in range(self.max_steps):
            print(f"\n--- Step {step + 1} ---")

            response = client.chat.completions.create(
                model="gpt-4o", messages=messages, temperature=0.3
            )
            reply = response.choices[0].message.content
            print(f"AI: {reply}")
            messages.append({"role": "assistant", "content": reply})

            # 检查最终答案
            if "FINAL_ANSWER:" in reply:
                return reply.split("FINAL_ANSWER:")[1].strip()

            # 检查工具调用
            if "TOOL_CALL:" in reply:
                tool_json = reply.split("TOOL_CALL:")[1].strip()
                try:
                    call = json.loads(tool_json)
                    tool_name = call["tool"]
                    tool_args = call.get("args", {})

                    print(f"🔧 调用工具: {tool_name}({tool_args})")
                    result = self.tools[tool_name]["func"](**tool_args)
                    print(f"📋 结果: {result}")

                    messages.append({
                        "role": "user",
                        "content": f"工具 {tool_name} 的执行结果：{result}"
                    })
                except Exception as e:
                    messages.append({
                        "role": "user", "content": f"工具调用失败：{e}"
                    })

        return "达到最大步骤数，任务未完成"

# ========== 使用 ==========
agent = SimpleAgent(TOOLS)
answer = agent.run("现在几点了？另外帮我算一下 1024 * 768 等于多少")
print(f"\n最终答案: {answer}")

📌 Agent 的核心就是一个 While 循环

  • ① LLM 思考，决定下一步做什么
  • ② 如果需要工具 → 调用工具 → 把结果喂回 LLM
  • ③ 如果不需要工具 → 输出最终答案 → 退出循环
  • ④ 设置 max_steps 防止无限循环

langchain_agent.py
from langchain_openai import ChatOpenAI
from langchain.agents import create_tool_calling_agent, AgentExecutor
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.tools import tool

# ---- 1. 定义工具（用装饰器，超简单）----
@tool
def calculator(expression: str) -> str:
    """计算数学表达式。输入一个数学表达式字符串，如 '2 + 3 * 4'"""
    try:
        return str(eval(expression, {"__builtins__": {}}, {}))
    except:
        return "计算出错"

@tool
def get_word_count(text: str) -> str:
    """统计文本的字数和字符数"""
    words = len(text.split())
    chars = len(text)
    return f"单词数: {words}, 字符数: {chars}"

tools = [calculator, get_word_count]

# ---- 2. 创建 Agent ----
llm = ChatOpenAI(model="gpt-4o", temperature=0.3)
prompt = ChatPromptTemplate.from_messages([
    ("system", "你是一个乐于助人的 AI 助手，可以使用工具完成任务。"),
    ("human", "{input}"),
    ("placeholder", "{agent_scratchpad}"),
])

agent = create_tool_calling_agent(llm, tools, prompt)
executor = AgentExecutor(agent=agent, tools=tools, verbose=True)

# ---- 3. 运行 ----
result = executor.invoke({"input": "计算 (125 * 37) + (89 * 43) 的结果"})
print(result["output"])

from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from typing import List

app = FastAPI(title="AI 知识库 API")

# ---- 请求/响应模型 ----
class QueryRequest(BaseModel):
    question: str
    top_k: int = 3

class QueryResponse(BaseModel):
    answer: str
    sources: List[str]
    tokens: int

# ---- API 路由 ----
@app.post("/ask", response_model=QueryResponse)
async def ask_question(req: QueryRequest):
    try:
        result = rag_query(req.question, top_k=req.top_k)
        return QueryResponse(
            answer=result["answer"],
            sources=result["sources"],
            tokens=result.get("tokens", 0)
        )
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.get("/health")
async def health():
    return {"status": "ok"}

# 启动: uvicorn api_server:app --reload

🤖 LLM-as-Judge（用 AI 评判 AI） 必做
llm_judge.py
JUDGE_PROMPT = """你是一个 AI 回答质量评判官。请评估以下回答的质量。

问题：{question}
参考答案：{expected}
实际回答：{actual}

请从以下维度打分（1-5）：
- 准确性：回答是否正确
- 完整性：是否包含关键信息
- 简洁性：是否简明扼要

输出 JSON：{{"accuracy": X, "completeness": X, "conciseness": X, "total": X, "comment": "..."}}"""

def judge_answer(question, expected, actual):
    resp = client.chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "user", "content": JUDGE_PROMPT.format(
            question=question, expected=expected, actual=actual
        )}],
        response_format={"type": "json_object"},
        temperature=0
    )
    return json.loads(resp.choices[0].message.content)
