🔤 字符串操作（重点！） 必做

在 AI 开发中，你大量时间都在处理文本，所以字符串操作是最常用的技能之一。

strings.py
text = "Hello, Python World!"

# ---- 基本操作 ----
print(len(text))             # 长度: 21
print(text.lower())          # 全小写: hello, python world!
print(text.upper())          # 全大写: HELLO, PYTHON WORLD!
print(text.replace("World", "AI"))  # 替换: Hello, Python AI!

# ---- 切片（超级常用）----
print(text[0])               # 第一个字符: H
print(text[7:13])            # 截取: Python
print(text[-6:])             # 倒数6个: orld!

# ---- 分割与拼接 ----
words = text.split(",")      # 按逗号分割: ['Hello', ' Python World!']
joined = " | ".join(["a", "b", "c"])  # 拼接: a | b | c

# ---- 查找与判断 ----
print("Python" in text)     # True（是否包含）
print(text.startswith("Hello"))  # True
print(text.find("Python"))   # 7（返回位置）

# ---- f-string 格式化（最推荐）----
name = "小明"
score = 95.6
print(f"{name} 的成绩是 {score:.1f} 分")  # 小明 的成绩是 95.6 分

  • "".join() — 把列表拼回字符串
  • f"..." — 格式化输出（拼接 prompt 时天天用）

🗂️ 字典 (Dict) — AI 开发的核心数据结构 必做

字典是键值对集合，用 {} 表示。API 返回的 JSON 数据本质就是字典，必须熟练掌握！

dicts.py
# ---- 创建字典 ----
person = {
    "name": "小明",
    "age": 25,
    "skills": ["Python", "SQL"],
    "is_student": False
}

# ---- 访问与修改 ----
print(person["name"])           # 小明
print(person.get("age"))       # 25（推荐，不存在时返回 None）
print(person.get("email", "无")) # "无"（不存在时返回默认值）

person["age"] = 26            # 修改
person["email"] = "a@b.com"  # 新增
del person["is_student"]       # 删除

# ---- 遍历字典 ----
for key, value in person.items():
    print(f"{key}: {value}")

# ---- 嵌套字典（模拟 API 返回值）----
api_response = {
    "status": "success",
    "data": {
        "message": "你好，世界",
        "tokens": 15
    }
}
# 取出嵌套的值
msg = api_response["data"]["message"]
print(msg)  # 你好，世界

📌 为什么字典这么重要？

  • 大模型 API 的请求和返回值全是 JSON 格式 = Python 字典
  • 配置文件、Prompt 模板管理都用字典
  • 后续学的 Pandas 的每一行数据本质也是键值对

📄 文件读写 必做
file_io.py
# ---- 写入文件 ----
with open("output.txt", "w", encoding="utf-8") as f:
    f.write("第一行内容\n")
    f.write("第二行内容\n")

# ---- 读取文件 ----
with open("output.txt", "r", encoding="utf-8") as f:
    content = f.read()          # 读全部内容
    print(content)

with open("output.txt", "r", encoding="utf-8") as f:
    lines = f.readlines()       # 按行读取，返回列表
    for line in lines:
        print(line.strip())     # strip() 去掉换行符

# ---- 追加内容 ----
with open("output.txt", "a", encoding="utf-8") as f:
    f.write("追加的第三行\n")

📌 with 语句的意义

  • with open() 会在代码块结束后自动关闭文件，不用手动 close()
  • 始终加 encoding="utf-8"，避免中文乱码
  • "w" = 写（覆盖），"r" = 读，"a" = 追加

📊 json — 数据交换格式 必做
use_json.py
import json

# ---- 字典 → JSON 字符串 ----
data = {"name": "小明", "scores": [85, 92, 78]}
json_str = json.dumps(data, ensure_ascii=False, indent=2)
print(json_str)

# ---- JSON 字符串 → 字典 ----
parsed = json.loads(json_str)
print(parsed["name"])         # 小明

# ---- 读写 JSON 文件 ----
with open("config.json", "w", encoding="utf-8") as f:
    json.dump(data, f, ensure_ascii=False, indent=2)

with open("config.json", "r", encoding="utf-8") as f:
    loaded = json.load(f)
    print(loaded)

💡 json 和 dict 的关系

JSON 是一种文本格式，Python 字典是内存中的数据结构。json.dumps() 把字典变成文本，json.loads() 把文本变回字典。API 通信全靠它们。

🐛 读懂报错信息 必做

不会调试 = 永远依赖别人。学会读报错信息是独立开发的分水岭。

常见报错类型
# ❌ SyntaxError — 语法写错了
# 看报错指向的行号，通常是漏了冒号、括号、引号
if x == 1     # 少了冒号 :

# ❌ NameError — 变量名不存在
print(username)  # 没有定义过 username

# ❌ TypeError — 类型不对
"age: " + 25    # 字符串不能和数字直接拼，要 str(25)

# ❌ KeyError — 字典里找不到这个 key
data = {"name": "小明"}
data["age"]     # 用 data.get("age") 更安全

# ❌ IndexError — 列表下标越界
lst = [1, 2, 3]
lst[5]           # 只有 0, 1, 2 三个下标

# ❌ FileNotFoundError — 找不到文件
open("不存在.txt")  # 检查路径和文件名

📌 调试三步法

  • 第一步：看报错的最后一行——告诉你错误类型和原因
  • 第二步：看报错中的行号——定位出错的位置
  • 第三步：把报错信息复制给 ChatGPT / Claude——AI 解释比自己查快 10 倍

import requests

def safe_api_call(url):
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()  # 非 200 抛异常
        return response.json()
    except requests.Timeout:
        print("请求超时")
    except requests.RequestException as e:
        print(f"请求失败: {e}")
    return None

import json
import os

def read_csv(filepath):
    """读取 CSV 文件，返回字典列表"""
    try:
        with open(filepath, "r", encoding="utf-8") as f:
            lines = f.readlines()
        headers = lines[0].strip().split(",")
        data = []
        for line in lines[1:]:
            values = line.strip().split(",")
            row = dict(zip(headers, values))
            data.append(row)
        return data
    except FileNotFoundError:
        print(f"文件不存在: {filepath}")
        return []

def filter_data(data, column, value):
    """按条件筛选数据"""
    return [row for row in data if row.get(column) == value]

def summarize(data, column):
    """对数值列求和、平均值"""
    values = [float(row[column]) for row in data if column in row]
    return {
        "count": len(values),
        "sum": sum(values),
        "average": sum(values) / len(values) if values else 0
    }

def save_result(data, filepath):
    """保存结果为 JSON"""
    with open(filepath, "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)
    print(f"结果已保存到 {filepath}")

# ---- 主流程 ----
if __name__ == "__main__":
    data = read_csv("students.csv")
    print(f"共读取 {len(data)} 条数据")

    # 筛选
    class_a = filter_data(data, "class", "A")
    print(f"A班学生: {len(class_a)} 人")

    # 统计
    stats = summarize(data, "score")
    print(f"平均分: {stats['average']:.1f}")

    # 保存
    save_result(stats, "result.json")
