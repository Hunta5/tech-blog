📊 Series — 一维数据（一列数据） 重要

Series 就像 Excel 里的"一列"，有数据也有索引。理解它是理解 DataFrame 的基础。

series.py
import pandas as pd

# ---- 从列表创建 Series ----
scores = pd.Series([85, 92, 78, 96, 88])
print(scores)
# 0    85
# 1    92
# 2    78
# 3    96
# 4    88

# ---- 自定义索引 ----
scores = pd.Series(
    [85, 92, 78, 96, 88],
    index=["小明", "小红", "小刚", "小李", "小王"]
)
print(scores["小红"])     # 92

# ---- 常用统计 ----
print(scores.mean())      # 平均值: 87.8
print(scores.max())       # 最大值: 96
print(scores.min())       # 最小值: 78
print(scores.std())       # 标准差
print(scores.describe())  # 一次性看全部统计信息

# ---- 向量化运算（不用 for 循环！）----
print(scores + 5)         # 每人加 5 分
print(scores > 90)        # 返回布尔 Series
print(scores[scores > 90]) # 筛选出 >90 的
📋 DataFrame — 二维表格（核心中的核心） 必做

DataFrame 就是一张表格，有行有列，就像 Excel 的一个 sheet。你在 Pandas 里 95% 的时间都在操作 DataFrame。

dataframe.py
import pandas as pd

# ---- 从字典创建（最常见方式）----
data = {
    "name":  ["小明", "小红", "小刚", "小李", "小王"],
    "age":   [22, 21, 23, 22, 24],
    "score": [85, 92, 78, 96, 88],
    "city":  ["北京", "上海", "北京", "深圳", "上海"]
}
df = pd.DataFrame(data)
print(df)
#   name  age  score city
# 0  小明   22     85  北京
# 1  小红   21     92  上海
# 2  小刚   23     78  北京
# 3  小李   22     96  深圳
# 4  小王   24     88  上海

# ---- 基本属性 ----
print(df.shape)        # (5, 4) → 5行4列
print(df.columns)      # Index(['name', 'age', 'score', 'city'])
print(df.dtypes)       # 每列的数据类型
print(df.info())       # 完整信息摘要

# ---- 取列（返回 Series）----
print(df["name"])       # 取一列
print(df[["name", "score"]])  # 取多列（双层中括号）

# ---- 取行 ----
print(df.head(3))       # 前 3 行
print(df.tail(2))       # 后 2 行

# ---- 快速统计 ----
print(df.describe())   # 数值列的统计摘要
print(df["score"].mean())  # 平均分
print(df["city"].value_counts())  # 每个城市出现几次

🔍 数据探索三板斧 必做

拿到任何数据，第一件事就是"探索"——看看有什么列、多少行、有没有缺失值、数据分布是啥样。

explore.py — 拿到数据后的标准流程
import pandas as pd

df = pd.read_csv("data.csv")

# ========== 第一斧：看结构 ==========
print(df.shape)            # 几行几列？
print(df.columns.tolist())  # 列名列表
print(df.dtypes)            # 每列数据类型
print(df.head())            # 看前 5 行长什么样

# ========== 第二斧：看缺失 ==========
print(df.isnull().sum())    # 每列有多少缺失值
print(df.isnull().sum() / len(df) * 100)  # 缺失率（%）

# ========== 第三斧：看分布 ==========
print(df.describe())           # 数值列统计
print(df["city"].value_counts())  # 分类列频次
print(df["score"].hist())         # 分布直方图（可选）

📌 养成习惯：每次读取数据后，先跑这三板斧

  • 看结构 → 知道有什么数据可以用
  • 看缺失 → 决定后续怎么清洗
  • 看分布 → 发现异常值和数据特征

🎯 条件筛选（最常用的操作，没有之一） 必做
filtering.py
import pandas as pd

df = pd.DataFrame({
    "name":  ["小明", "小红", "小刚", "小李", "小王"],
    "age":   [22, 21, 23, 22, 24],
    "score": [85, 92, 78, 96, 88],
    "city":  ["北京", "上海", "北京", "深圳", "上海"]
})

# ---- 单条件筛选 ----
high_score = df[df["score"] > 90]           # 成绩 > 90
beijing = df[df["city"] == "北京"]          # 北京的同学

# ---- 多条件筛选（& 且 | 或）----
result = df[(df["score"] > 80) & (df["city"] == "上海")]
result = df[(df["city"] == "北京") | (df["city"] == "上海")]

# ---- isin() 多值匹配 ----
big_city = df[df["city"].isin(["北京", "上海"])]

# ---- 字符串筛选 ----
df[df["name"].str.contains("小")]    # 名字包含"小"
df[df["name"].str.startswith("小")]  # 名字以"小"开头

# ---- between() 范围筛选 ----
df[df["score"].between(80, 95)]       # 80 ≤ score ≤ 95

⚠️ 多条件筛选必须加括号！

错误写法：df[df["score"] > 80 & df["city"] == "上海"]
正确写法：df[(df["score"] > 80) & (df["city"] == "上海")]
因为 & 的优先级高于 >，不加括号会报错。

📍 loc 和 iloc — 精准定位 必做
loc_iloc.py
# ---- loc：按标签（名称）选取 ----
df.loc[0, "name"]              # 第 0 行的 name → "小明"
df.loc[0:2, ["name", "score"]]  # 0~2 行的 name 和 score
df.loc[df["score"] > 90, "name"]  # 成绩>90 的人的名字

# ---- iloc：按位置（数字）选取 ----
df.iloc[0, 0]         # 第 0 行第 0 列
df.iloc[0:3, 0:2]     # 前 3 行，前 2 列
df.iloc[-1]            # 最后一行

# ---- 修改值 ----
df.loc[0, "score"] = 90    # 修改小明的成绩
df.loc[df["city"] == "北京", "score"] += 5  # 北京的加 5 分

📌 loc vs iloc 一句话区分

  • loc → 用名字找（label-based），支持条件筛选
  • iloc → 用位置号找（integer-based），纯数字下标

🕳️ 处理缺失值 (NaN) 必做
missing_values.py
import pandas as pd
import numpy as np

df = pd.DataFrame({
    "name":  ["小明", "小红", "小刚", np.nan, "小王"],
    "score": [85, np.nan, 78, 96, np.nan],
    "city":  ["北京", "上海", np.nan, "深圳", "上海"]
})

# ---- 检查缺失 ----
print(df.isnull())             # 每个格子是否为空
print(df.isnull().sum())        # 每列缺失数量

# ---- 删除缺失 ----
df.dropna()                     # 删除含缺失值的行
df.dropna(subset=["name"])      # 只删 name 为空的行
df.dropna(thresh=2)             # 至少有 2 个非空值才保留

# ---- 填充缺失 ----
df["score"].fillna(0)           # 填 0
df["score"].fillna(df["score"].mean())  # 填平均值（常用）
df["city"].fillna("未知")       # 填自定义文本
df.fillna(method="ffill")       # 用上一行的值填充

🔄 类型转换与文本清洗 必做
type_cleaning.py
# ---- 类型转换 ----
df["score"] = df["score"].astype(float)      # 转浮点数
df["age"] = pd.to_numeric(df["age"], errors="coerce")  # 安全转数字，无法转的变 NaN

DAY9-10
分组聚合与透视表
groupby 是 Pandas 最强大的功能——"按XX分组，统计YY"
📊 groupby — 分组聚合 必做

groupby 的思维模式：拆分 → 应用 → 合并。先按某列分组，再对每组做统计，最后合并结果。

groupby.py
import pandas as pd

df = pd.DataFrame({
    "name":     ["小明","小红","小刚","小李","小王","小张"],
    "city":     ["北京","上海","北京","深圳","上海","北京"],
    "department": ["技术","技术","产品","技术","产品","技术"],
    "salary":   [15000,18000,12000,20000,13000,16000],
    "age":      [25,28,30,26,32,24]
})

# ---- 按城市分组，算平均工资 ----
print(df.groupby("city")["salary"].mean())
# city
# 上海    15500
# 北京    14333
# 深圳    20000

# ---- 多个聚合函数 ----
print(df.groupby("city")["salary"].agg(["mean", "max", "min", "count"]))

# ---- 多列分组 ----
print(df.groupby(["city", "department"])["salary"].mean())

# ---- agg() 对不同列做不同统计 ----
result = df.groupby("city").agg({
    "salary": ["mean", "max"],
    "age": "mean",
    "name": "count"       # 计数
})
print(result)

# ---- 分组后排序 ----
city_salary = df.groupby("city")["salary"].mean().sort_values(ascending=False)
print(city_salary)
📐 pivot_table — 透视表 重要

透视表就是 Excel 里的"数据透视表"功能——行列交叉统计，一行代码搞定。

    index="city",             # 行

)
print(pivot)

# city
# 上海       13000.0  18000.0
# 北京       12000.0  15500.0
# 深圳           NaN  20000.0

# ---- 添加行/列汇总 ----
pivot = df.pivot_table(

多表合并与连接
真实项目数据分散在多个表——merge 帮你拼起来
🔗 merge — 表连接（类似 SQL 的 JOIN） 必做
merge.py
import pandas as pd

# 两张表
users = pd.DataFrame({
    "user_id": [1, 2, 3, 4],
    "name": ["小明", "小红", "小刚", "小李"]
})
orders = pd.DataFrame({
    "order_id": [101, 102, 103, 104],
    "user_id": [1, 2, 2, 5],
    "amount": [100, 200, 150, 300]
})

# ---- inner join（只保留两边都有的）----
result = pd.merge(users, orders, on="user_id", how="inner")

# ---- left join（保留左表全部）----
result = pd.merge(users, orders, on="user_id", how="left")

# ---- right join（保留右表全部）----
result = pd.merge(users, orders, on="user_id", how="right")

# ---- outer join（两边全保留）----
result = pd.merge(users, orders, on="user_id", how="outer")
连接方式保留数据类比
inner两边都有的取交集
left左表全部 + 右表匹配以左表为主
right右表全部 + 左表匹配以右表为主
outer两边全部取并集
📎 concat — 上下/左右拼接 重要
concat.py
# ---- 上下拼接（追加行）----
df1 = pd.DataFrame({"name": ["A", "B"], "score": [80, 90]})
df2 = pd.DataFrame({"name": ["C", "D"], "score": [85, 95]})

combined = pd.concat([df1, df2], ignore_index=True)
# ignore_index=True 重新编号索引

# ---- 左右拼接（追加列）----
info = pd.DataFrame({"city": ["北京", "上海"]})
combined = pd.concat([df1, info], axis=1)

# ---- 实际场景：合并多个 CSV ----
import glob
files = glob.glob("data/*.csv")
all_dfs = [pd.read_csv(f) for f in files]
merged = pd.concat(all_dfs, ignore_index=True)

💡 merge vs concat

merge = 按共同列关联（像 SQL JOIN）
concat = 简单堆叠（上下追加行 或 左右追加列）

DAY13-14
SQL 基础查询
数据库是企业数据的大本营——学会 SELECT 就够用了
🗄️ SQL 核心语法 必做

SQL 的核心就一个模式：从哪张表（FROM）选哪些列（SELECT），满足什么条件（WHERE）。

SQL 基础语法
-- ---- 基本查询 ----
SELECT * FROM users;                       -- 查全部
SELECT name, age FROM users;               -- 选列
SELECT * FROM users WHERE age > 20;       -- 条件筛选
SELECT * FROM users WHERE city = '北京';  -- 文字条件

-- ---- AND / OR / IN ----
SELECT * FROM users
WHERE age > 20 AND city = '北京';

SELECT * FROM users
WHERE city IN ('北京', '上海');

-- ---- 排序 ----
SELECT * FROM users ORDER BY age DESC;   -- 降序
SELECT * FROM users ORDER BY age ASC;    -- 升序（默认）

-- ---- LIMIT ----
SELECT * FROM users ORDER BY score DESC LIMIT 10;  -- 前 10 名

-- ---- 去重 ----
SELECT DISTINCT city FROM users;           -- 有哪些城市
📈 GROUP BY 与聚合函数 必做
SQL 分组聚合
-- ---- 聚合函数 ----
SELECT COUNT(*) FROM users;               -- 总行数
SELECT AVG(score) FROM users;              -- 平均分
SELECT MAX(score), MIN(score) FROM users;  -- 最高最低分
SELECT SUM(salary) FROM users;             -- 工资总和

-- ---- GROUP BY 分组 ----
SELECT city, AVG(salary) AS avg_salary
FROM users
GROUP BY city;

-- ---- HAVING 对分组结果筛选 ----
SELECT city, COUNT(*) AS cnt
FROM users
GROUP BY city
HAVING cnt > 2;                     -- 人数 > 2 的城市
🔗 JOIN 表连接 必做
SQL JOIN
-- ---- INNER JOIN ----
SELECT u.name, o.amount
FROM users u
INNER JOIN orders o ON u.user_id = o.user_id;

-- ---- LEFT JOIN ----
SELECT u.name, o.amount
FROM users u
LEFT JOIN orders o ON u.user_id = o.user_id;
-- 保留所有用户，没订单的 amount 为 NULL
🐍 Python 中执行 SQL（用 SQLite 练习） 必做
sql_in_python.py
import sqlite3
import pandas as pd

# ---- 创建内存数据库 + 导入数据 ----
conn = sqlite3.connect(":memory:")

df = pd.DataFrame({
    "name": ["小明","小红","小刚"],
    "city": ["北京","上海","北京"],
    "score": [85, 92, 78]
})
df.to_sql("students", conn, index=False)

# ---- 用 SQL 查询，结果直接变 DataFrame ----
result = pd.read_sql("SELECT * FROM students WHERE score > 80", conn)
print(result)

# ---- 分组统计 ----
result = pd.read_sql("""
    SELECT city, AVG(score) AS avg_score, COUNT(*) AS cnt
    FROM students
    GROUP BY city
""", conn)
print(result)

conn.close()

import json
import pandas as pd

# 模拟 AI API 返回值
api_response = {
    "id": "chatcmpl-abc123",
    "model": "gpt-4",
    "choices": [
        {
            "index": 0,
            "message": {
                "role": "assistant",
                "content": "Python 是一种编程语言。"
            },
            "finish_reason": "stop"
        }
    ],
    "usage": {
        "prompt_tokens": 10,
        "completion_tokens": 15,
        "total_tokens": 25
    }
}

# ---- 逐层提取 ----
content = api_response["choices"][0]["message"]["content"]
tokens = api_response["usage"]["total_tokens"]

print(f"消耗 token: {tokens}")

# ---- 批量 API 结果 → DataFrame ----
results = [
    {"question": "Q1", "answer": "A1", "tokens": 20},
    {"question": "Q2", "answer": "A2", "tokens": 35},
    {"question": "Q3", "answer": "A3", "tokens": 18},
]
df = pd.DataFrame(results)

print(f"平均 token: {df['tokens'].mean():.1f}")

# ---- json_normalize：自动展平嵌套 JSON ----
nested_data = [
    {"name": "小明", "info": {"city": "北京", "age": 25}},
    {"name": "小红", "info": {"city": "上海", "age": 28}},
]

print(df)

import pandas as pd
import json
from datetime import datetime

# ---- Step 1: 加载模拟数据 ----
chat_logs = [
    {"user_id":"u1", "time":"2026-03-28 10:00", "role":"user",
     "content":"Python怎么读CSV？", "tokens":8},
    {"user_id":"u1", "time":"2026-03-28 10:00", "role":"assistant",
     "content":"用 pandas 的 read_csv...", "tokens":120},
    {"user_id":"u2", "time":"2026-03-28 11:30", "role":"user",
     "content":"帮我写个爬虫", "tokens":6},
    # ... 更多数据
]

df = pd.DataFrame(chat_logs)
df["time"] = pd.to_datetime(df["time"])

# ---- Step 2: 数据清洗 ----
df["content"] = df["content"].str.strip()
df = df.dropna(subset=["content"])
df["hour"] = df["time"].dt.hour

# ---- Step 3: 统计分析 ----
# 每个用户的对话次数
user_stats = df[df["role"]=="user"].groupby("user_id").agg(
    对话次数=("content", "count"),
    总token=("tokens", "sum")
).reset_index()

# AI 回复的 token 统计
ai_stats = df[df["role"]=="assistant"]["tokens"].describe()

# 按小时统计活跃度
hourly = df.groupby("hour").size()

# ---- Step 4: 导出报告 ----
report = {
    "总对话数": len(df),
    "独立用户数": df["user_id"].nunique(),
    "总token消耗": int(df["tokens"].sum()),
    "平均回复token": round(float(ai_stats["mean"]), 1),
    "生成时间": datetime.now().strftime("%Y-%m-%d %H:%M")
}

with open("chat_report.json", "w", encoding="utf-8") as f:
    json.dump(report, f, ensure_ascii=False, indent=2)

user_stats.to_csv("user_stats.csv", index=False)
print("报告生成完毕！")
