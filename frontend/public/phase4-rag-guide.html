💡 什么是 Embedding？ 必做

Embedding 就是把文字变成一串数字（向量）。意思相近的文字，向量也相近。这样计算机就能"理解"语义了——不是靠关键词匹配，而是靠意思相似度。

文本向量（简化示意）说明
"我喜欢猫"[0.82, -0.15, 0.43, ...]意思相近 → 向量接近
"我爱小猫咪"[0.80, -0.12, 0.45, ...]
"今天股票涨了"[-0.31, 0.67, -0.22, ...]意思不同 → 向量远离

from openai import OpenAI
import numpy as np

client = OpenAI()

# ---- 单条文本向量化 ----
response = client.embeddings.create(
    model="text-embedding-3-small",   # 推荐，便宜又好用
    input="Python 是一种编程语言"
)
vector = response.data[0].embedding
print(f"向量维度: {len(vector)}")      # 1536
print(f"前5个值: {vector[:5]}")

# ---- 批量向量化 ----
texts = [
    "Python 是一种编程语言",
    "Java 也是编程语言",
    "今天天气真好",
]
response = client.embeddings.create(
    model="text-embedding-3-small",
    input=texts
)
vectors = [d.embedding for d in response.data]

# ---- 计算相似度 ----
def cosine_similarity(a, b):
    a, b = np.array(a), np.array(b)
    return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))

print(f"Python vs Java: {cosine_similarity(vectors[0], vectors[1]):.4f}")
# → 0.89（很相似！都是编程语言）

print(f"Python vs 天气: {cosine_similarity(vectors[0], vectors[2]):.4f}")
# → 0.21（不相似，话题完全不同）

# ---- 读取 TXT ----
def load_txt(path):
    with open(path, "r", encoding="utf-8") as f:
        return f.read()

# ---- 读取 PDF ----
from pypdf import PdfReader

def load_pdf(path):
    reader = PdfReader(path)
    text = ""
    for page in reader.pages:
        text += page.extract_text() + "\n"
    return text

# ---- 读取 Word ----
from docx import Document

def load_docx(path):
    doc = Document(path)
    return "\n".join([p.text for p in doc.paragraphs])

# ---- 统一加载接口 ----
def load_document(path):
    if path.endswith(".pdf"):
        return load_pdf(path)
    elif path.endswith(".docx"):
        return load_docx(path)
    elif path.endswith(".txt") or path.endswith(".md"):
        return load_txt(path)
    else:
        raise ValueError(f"不支持的文件格式: {path}")
✂️ 文本切分（Chunking）— RAG 最关键的步骤 必做

一篇长文档不能整体做 Embedding（太长了），必须切成小块（chunk）。切分策略直接决定 RAG 的检索质量——切太大找不准，切太小丢上下文。

chunking.py
# ==== 方法 1: 固定长度 + 重叠（最常用）====
def chunk_by_size(text, chunk_size=500, overlap=100):
    """按字符数切分，相邻块有重叠"""
    chunks = []
    start = 0
    while start < len(text):
        end = start + chunk_size
        chunk = text[start:end]
        chunks.append(chunk)
        start = end - overlap   # 重叠部分保持上下文连贯
    return chunks

# ==== 方法 2: 按段落切分 ====
def chunk_by_paragraph(text, max_size=800):
    """按段落切分，短段落合并，长段落拆分"""
    paragraphs = text.split("\n\n")
    chunks = []
    current = ""
    for para in paragraphs:
        if len(current) + len(para) < max_size:
            current += para + "\n\n"
        else:
            if current:
                chunks.append(current.strip())
            current = para + "\n\n"
    if current.strip():
        chunks.append(current.strip())
    return chunks

# ==== 方法 3: 使用 LangChain 的 Splitter（推荐）====
from langchain.text_splitter import RecursiveCharacterTextSplitter

splitter = RecursiveCharacterTextSplitter(
    chunk_size=500,        # 每块最大字符数
    chunk_overlap=100,     # 重叠字符数
    separators=["\n\n", "\n", "。", "！", "？", "，", " "]
    # 优先按段落分，其次按句子，最后按字符
)

text = load_document("product_manual.pdf")
chunks = splitter.split_text(text)
print(f"共切分为 {len(chunks)} 个块")
print(f"平均长度: {sum(len(c) for c in chunks) / len(chunks):.0f} 字符")

📌 切分经验法则

  • chunk_size 500-1000 字符通常效果最好
  • overlap 10%-20% 保持跨块的上下文连贯性
  • 中文按句号切比按空格切效果好
  • 表格数据建议每行为一个 chunk，加上表头作为上下文
🏷️ 添加元数据（Metadata） 重要

每个 chunk 不只有文本内容，还应该带上"来自哪个文件、第几页、什么类别"等信息。回答时可以引用溯源。

metadata.py
def create_documents(file_path, chunks):
    """为每个 chunk 添加元数据"""
    documents = []
    for i, chunk in enumerate(chunks):
        doc = {
            "id": f"{file_path}_chunk_{i}",
            "text": chunk,
            "metadata": {
                "source": file_path,
                "chunk_index": i,
                "total_chunks": len(chunks),
                "char_count": len(chunk),
            }
        }
        documents.append(doc)
    return documents

📁 完整的文档入库管线 必做
indexing_pipeline.py — 文件 → 切分 → 向量化 → 存储
import os, glob

def index_documents(folder_path, collection, chunk_size=500, overlap=100):
    """将文件夹中所有文档建立索引"""
    from langchain.text_splitter import RecursiveCharacterTextSplitter
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=chunk_size, chunk_overlap=overlap,
        separators=["\n\n", "\n", "。", "！", "？", " "]
    )

    all_chunks, all_ids, all_metas = [], [], []
    doc_count = 0

    for filepath in glob.glob(os.path.join(folder_path, "*")):
        try:
            text = load_document(filepath)
            chunks = splitter.split_text(text)
            filename = os.path.basename(filepath)

            for i, chunk in enumerate(chunks):
                all_chunks.append(chunk)
                all_ids.append(f"{filename}_{i}")
                all_metas.append({
                    "source": filename,
                    "chunk_index": i
                })

            doc_count += 1
            print(f"✅ {filename}: {len(chunks)} 个块")
        except Exception as e:
            print(f"❌ {filepath}: {e}")

    # 批量向量化并存入
    if all_chunks:
        embeddings = get_embeddings(all_chunks)
        collection.add(
            ids=all_ids,
            documents=all_chunks,
            embeddings=embeddings,
            metadatas=all_metas
        )

    print(f"\n完成！共处理 {doc_count} 个文件，{len(all_chunks)} 个块")

# 使用
index_documents("./docs", collection)

🔗 RAG 完整流程实现 必做
rag_pipeline.py — 完整的 RAG 问答
from openai import OpenAI
import chromadb

client = OpenAI()
chroma = chromadb.PersistentClient(path="./my_vectordb")
collection = chroma.get_collection("knowledge_base")

RAG_PROMPT = """你是一个智能问答助手。请根据以下参考资料回答用户的问题。

规则：
- 只根据参考资料回答，不要编造信息
- 如果参考资料中没有相关信息，请说"根据现有资料，我无法回答这个问题"
- 在回答末尾标注信息来源

参考资料：
---
{context}
---

用户问题：{question}"""

def rag_query(question, top_k=3):
    """完整的 RAG 问答流程"""

    # Step 1: 检索相关文档
    query_emb = get_embeddings([question])[0]
    results = collection.query(
        query_embeddings=[query_emb],
        n_results=top_k,
        include=["documents", "metadatas", "distances"]
    )

    # Step 2: 拼装上下文
    context_parts = []
    sources = []
    for i, doc in enumerate(results["documents"][0]):
        source = results["metadatas"][0][i].get("source", "未知")
        context_parts.append(f"[来源: {source}]\n{doc}")
        sources.append(source)

    context = "\n\n".join(context_parts)

    # Step 3: 调用大模型生成回答
    prompt = RAG_PROMPT.format(context=context, question=question)
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "user", "content": prompt}],
        temperature=0.3    # RAG 场景建议低温度，更忠实
    )
    answer = response.choices[0].message.content

    return {
        "answer": answer,
        "sources": list(set(sources)),
        "context_used": context_parts
    }

# ---- 使用 ----
result = rag_query("什么是 RAG？")
print(f"回答: {result['answer']}")
print(f"来源: {result['sources']}")

📌 RAG 的三个关键参数

  • top_k — 检索几条结果（通常 3-5 条，太多会干扰模型）
  • temperature — RAG 场景用 0-0.3，避免 AI "发挥创造力"编造信息
  • Prompt 模板 — 明确告诉 AI "只根据资料回答"，是防止幻觉的关键

retrieval_optimization.py
# ==== 优化 1: Query 改写 ====
# 用户问题口语化，先让 AI 改写成更适合检索的形式
def rewrite_query(original_query):
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{
            "role": "user",
            "content": f"""将以下用户问题改写为更适合知识库检索的形式。
输出 3 个不同角度的检索查询，每行一个。

用户问题：{original_query}"""
        }],
        temperature=0.3
    )
    queries = response.choices[0].message.content.strip().split("\n")
    return [q.strip() for q in queries if q.strip()]

# 例: "这东西怎么退" → ["退货流程", "退换货政策", "如何申请退货"]

# ==== 优化 2: 混合搜索 ====
# 同时用语义搜索 + 关键词搜索，取并集
def hybrid_search(query, collection, top_k=5):
    # 语义搜索
    emb = get_embeddings([query])[0]
    semantic_results = collection.query(
        query_embeddings=[emb], n_results=top_k
    )

    # 关键词搜索（Chroma 支持 where_document 过滤）
    keyword_results = collection.query(
        query_texts=[query], n_results=top_k  # 内置关键词匹配
    )

    # 合并去重
    seen = set()
    combined = []
    for source in [semantic_results, keyword_results]:
        for doc, doc_id in zip(source["documents"][0], source["ids"][0]):
            if doc_id not in seen:
                seen.add(doc_id)
                combined.append(doc)
    return combined[:top_k]

# ==== 优化 3: Rerank 重排序 ====
# 检索出 10 条，让 AI 挑出最相关的 3 条
def rerank(query, documents, top_k=3):
    doc_list = "\n".join([f"[{i}] {d[:200]}" for i, d in enumerate(documents)])
    resp = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": f"""以下文档中，哪些与问题最相关？
返回最相关的 {top_k} 个文档编号，用逗号分隔。

问题：{query}

文档列表：
{doc_list}"""}],
        temperature=0
    )
    indices = [int(x.strip()) for x in resp.choices[0].message.content.split(",")]
    return [documents[i] for i in indices if i < len(documents)]

# 防幻觉 Prompt 模板
ANTI_HALLUCINATION_PROMPT = """根据以下参考资料回答问题。

重要规则：
1. 只使用参考资料中的信息回答
2. 如果资料中没有答案，明确说"资料中未提及"
3. 直接引用原文关键句，用「」标注
4. 给出置信度（高/中/低）

参考资料：
{context}

问题：{question}

请按以下格式输出：
回答：...
引用：「原文句子」
置信度：高/中/低
来源：文件名"""

# 相似度阈值过滤——太不相关的结果直接丢弃
def filtered_search(query, collection, top_k=5, threshold=0.3):
    emb = get_embeddings([query])[0]
    results = collection.query(
        query_embeddings=[emb], n_results=top_k,
        include=["documents", "distances", "metadatas"]
    )
    # 只保留距离小于阈值的结果
    filtered = []
    for doc, dist, meta in zip(
        results["documents"][0],
        results["distances"][0],
        results["metadatas"][0]
    ):
        if dist < threshold:
            filtered.append({"doc": doc, "distance": dist, "meta": meta})
    return filtered

llm = ChatOpenAI(model="gpt-4o", temperature=0.3)
qa_chain = RetrievalQA.from_chain_type(
    llm=llm,
    retriever=vectordb.as_retriever(search_kwargs={"k": 3}),
    return_source_documents=True
)

# ---- 5. 提问 ----
result = qa_chain.invoke({"query": "什么是 RAG？"})
print(f"回答: {result['result']}")
for doc in result["source_documents"]:
    print(f"  来源: {doc.metadata['source']} (page {doc.metadata.get('page', '?')})")

knowledge_base_qa.py — 企业知识库问答系统
import os, json, glob
from openai import OpenAI
import chromadb
from langchain.text_splitter import RecursiveCharacterTextSplitter
from pypdf import PdfReader
from dotenv import load_dotenv

load_dotenv()
client = OpenAI()

class KnowledgeBaseQA:
    """企业知识库问答系统"""

    def __init__(self, db_path="./knowledge_db"):
        self.chroma = chromadb.PersistentClient(path=db_path)
        self.collection = self.chroma.get_or_create_collection("enterprise_kb")
        self.splitter = RecursiveCharacterTextSplitter(
            chunk_size=500, chunk_overlap=100,
            separators=["\n\n", "\n", "。", "！", "？", " "]
        )
        self.chat_history = []

    # ========== 文档管理 ==========
    def add_document(self, file_path):
        """添加单个文档到知识库"""
        text = self._load_file(file_path)
        chunks = self.splitter.split_text(text)
        filename = os.path.basename(file_path)

        ids = [f"{filename}_{i}" for i in range(len(chunks))]
        metas = [{"source": filename, "chunk_idx": i} for i in range(len(chunks))]
        embs = self._embed(chunks)

        self.collection.add(ids=ids, documents=chunks,
                           embeddings=embs, metadatas=metas)
        print(f"✅ {filename}: {len(chunks)} 块已入库")

    def add_folder(self, folder_path):
        """批量添加文件夹中的文档"""
        for f in glob.glob(os.path.join(folder_path, "*")):
            try:
                self.add_document(f)
            except Exception as e:
                print(f"❌ {f}: {e}")

    # ========== 智能问答 ==========
    def ask(self, question, top_k=3):
        """RAG 问答"""
        # 1. 检索
        emb = self._embed([question])[0]
        results = self.collection.query(
            query_embeddings=[emb], n_results=top_k,
            include=["documents", "metadatas", "distances"]
        )

        # 2. 过滤低质量结果
        context_parts, sources = [], []
        for doc, dist, meta in zip(
            results["documents"][0],
            results["distances"][0],
            results["metadatas"][0]
        ):
            if dist < 0.5:
                context_parts.append(doc)
                sources.append(meta["source"])

        if not context_parts:
            return {"answer": "抱歉，知识库中未找到相关信息。", "sources": []}

        # 3. 生成回答
        context = "\n---\n".join(context_parts)
        prompt = f"""根据以下参考资料回答问题。只基于资料回答，无法回答请说明。

参考资料：
{context}

问题：{question}"""

        response = client.chat.completions.create(
            model="gpt-4o",
            messages=[
                {"role": "system", "content": "你是企业知识库助手，基于提供的资料准确回答。"},
                {"role": "user", "content": prompt}
            ],
            temperature=0.2
        )

        return {
            "answer": response.choices[0].message.content,
            "sources": list(set(sources)),
            "tokens": response.usage.total_tokens
        }

    def stats(self):
        """查看知识库统计"""
        return {"total_chunks": self.collection.count()}

    # ========== 内部方法 ==========
    def _embed(self, texts):
        resp = client.embeddings.create(
            model="text-embedding-3-small", input=texts)
        return [d.embedding for d in resp.data]

    def _load_file(self, path):
        if path.endswith(".pdf"):
            reader = PdfReader(path)
            return "\n".join(p.extract_text() for p in reader.pages)
        else:
            with open(path, "r", encoding="utf-8") as f:
                return f.read()

# ========== 主程序 ==========
if __name__ == "__main__":
    kb = KnowledgeBaseQA()

    # 添加文档
    kb.add_folder("./docs")
    print(f"\n知识库状态: {kb.stats()}")

    # 交互式问答
    print("\n企业知识库问答系统已启动！输入 quit 退出。\n")
    while True:
        q = input("📌 你的问题: ")
        if q.lower() == "quit": break
        result = kb.ask(q)
        print(f"\n💬 回答: {result['answer']}")
        print(f"📎 来源: {', '.join(result['sources'])}")
        print(f"💰 Token: {result.get('tokens', '?')}\n")
